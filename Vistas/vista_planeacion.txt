CREATE OR REPLACE VIEW vista_planeacion AS SELECT planeacion_id AS vista_planeacion_id,planeacion_programa,programa_nombre,planeacion_ap,ap_nombre,fase,competencia_nombre,planeacion_rap,rap_nombre,planeacion_aa FROM planeacion,programa,fases,actividad_proyecto,competencias,rap WHERE (planeacion_programa = programa_id AND planeacion_ap=actividad_proyecto_id AND planeacion_rap=rap_id AND rap_competencia=competencias_id AND fases_id=ap_fase)
