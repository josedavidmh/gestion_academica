CREATE OR REPLACE VIEW vista_horario AS SELECT horario_id AS vista_horario_id,horario_dia,horario_instructor,instructor_numid,instructor_nombres,instructor_perfil,horario_fecha_inicio,horario_fecha_final,horario_hora_inicio,horario_hora_final,horario_ficha,ficha_nombre,vista_fichas.programa_nombre,vista_fichas.programa_codigo,ap_nombre,fase,horario_bloque,rap1,rap2,rap3,rap4,rap_codigo,rap_nombre,rap_horas_directas,rap_horas_otras,rap_conceptos,competencia_nombre,competencia_perfil FROM horario,vista_instructores,vista_planeacion_completa,vista_fichas WHERE horario_instructor=vista_instructores_id AND horario_ficha=vista_fichas_id AND (rap1=vista_planeacion_completa_id OR rap2=vista_planeacion_completa_id OR rap3=vista_planeacion_completa_id OR rap4=vista_planeacion_completa_id)
