CREATE OR REPLACE VIEW vista_instructores AS SELECT instructores_id AS vista_instructores_id,instructor_numid,instructor_nombres,instructor_correo,instructor_perfil,profesion_id,profesion_nombre,postgrado_id,CONCAT(postgrado_titulo," ",postgrado_nombre) postgrado FROM instructores,profesion,postgrado WHERE (instructor_profesion = profesion_id AND instructor_especialidad=postgrado_id)
